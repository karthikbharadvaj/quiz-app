async deleteApplicantIdentificationData(): Promise<void> {
  // Fetch the reference number for the user
  const referenceNumber = await this.#referenceNumberRepository.getReferenceNumber(this.#userUid);

  if (!referenceNumber) {
    throw new NotFoundError({ errorDetail: 'User has no application reference number' });
  }

  // Execute all deletion operations concurrently
  await Promise.all([
    this.#identificationAftConnectionDataRepository.deleteIdentificationAftConnectionData(referenceNumber, 'AGENT'),
    this.#identificationAftConnectionDataRepository.deleteIdentificationAftConnectionData(referenceNumber, 'REPRESENTATIVE'),
    this.#identificationDataRepository.deleteIdentificationData(referenceNumber, 'AGENT'),
    this.#identificationDataRepository.deleteIdentificationData(referenceNumber, 'REPRESENTATIVE'),
    this.#applicationRepository.saveValidationStatus({
      referenceNumber,
      key: 'isIdentificationApplicationDataValid',
      isValid: false,
    }),
  ]);
}
