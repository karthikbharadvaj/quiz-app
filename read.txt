import React from 'react';
import { User } from '../models/User';

interface UserCardProps {
  user: User;
  isSelected: boolean;
  onSelect: () => void;
}

const UserCard: React.FC<UserCardProps> = ({ user, isSelected, onSelect }) => {
  return (
    <div
      className={`user-card ${isSelected ? 'selected' : ''}`}
      onClick={onSelect}
    >
      <h2>{user.name}</h2>
    </div>
  );
};

export default UserCard;
