import React from "react";
import User, { IUser } from "./User";

interface UserCardProps {
  user: IUser;
  isSelected: boolean;
  onUserCardClick: (userId: number) => void;
}

const UserCard: React.FC<UserCardProps> = (props) => {
  const handleClick = (): void => {
    props.onUserCardClick(props.user.id);
  };

  const cardStyles = `rounded-lg p-4 border-2 ${
    props.isSelected ? "border-blue-500" : "border-gray-200"
  }`;

  return (
    <div className={cardStyles} onClick={handleClick}>
      <h2 className="text-xl font-medium">{props.user.name}</h2>
    </div>
  );
};

export default UserCard;
