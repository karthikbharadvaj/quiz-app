  const handleUserSelect = async (user: User) => {
    if (selectedUser === user) {
      setSelectedUser(null);
    } else {
      setIsLoading(true);
      try {
        const userDetails = await fetchUserDetails(user.id);
        setSelectedUser(userDetails);
      } catch (error) {
        setError(error.message);
      } finally {
        setIsLoading(false);
      }
    }
  };
